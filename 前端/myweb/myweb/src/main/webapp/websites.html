$.ajax({

    var phone = '您的注册手机号'; /*用您第一次提交的手机号替换"您的注册手机号"*/

    url: 'http://101.200.40.98:9090//url/my/' + phone, 

    type: 'GET',

    async: true,    /*是否异步, true是， false否*/

    data: { 

    },

    timeout: 1000 * 30,    /*超时时间，30s*/

    dataType: 'json',    /*返回的数据格式：json/xml/html/script/jsonp/text*/

    beforeSend: function(xhr){

        // loading

    },

    success: function(data, textStatus, jqXHR){

    // loaded


        /* data 为返回的数据，您可以对他进行操作*/

        /* 解析json数据示例*/

        // console.log(data.code + " " + data.msg + " " + data.data);

        var code = data.code;

        if (code == 0) /*code 为0时为正确返回，其他为错误代码*/

        {

        $.each(data.data, function(i, item) {

        item.name // 网站名称

        item.http // 网址

        item.time // 提交时间

        });

        } else {

        // Error : Msg

        }

    },

    error: function(xhr, textStatus){

    // loaded

    // a some error info

    },

    complete: function(){

        // completed

    }

});
